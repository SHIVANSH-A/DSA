Diff of elements at even and odd
// Online C++ compiler to run C++ program online
#include <bits/stdc++.h>
using namespace std;

int main() {
   vector<int> v ={1,2,3,4,5};
   int sum = 0;
   for(int i=0;i<5;i++){
       if(i%2==0){
           sum+=v[i];
       }
       else {
           sum-=v[i];
       }
   }
   cout<<sum;

    return 0;
}
